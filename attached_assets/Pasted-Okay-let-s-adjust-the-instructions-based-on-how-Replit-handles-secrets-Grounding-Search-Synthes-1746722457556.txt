Okay, let's adjust the instructions based on how Replit handles secrets.

**Grounding Search Synthesis on Replit Secrets:**

Replit uses a dedicated **Secrets** tool (found in the left sidebar under "Tools") for managing sensitive information like API keys. Users define secrets as key-value pairs within this tool. When the Repl runs, Replit automatically injects these secrets as environment variables (`process.env` in Node.js). This method is more secure than storing keys directly in `.env` files within the project's file system, as the Secrets tool keeps them hidden from the Repl's code editor and version control.

// Define the target structured output (can be moved to types.ts later)
interface StructuredLLMOutput {
  main_response: string;
  identified_topics?: string[];
  key_entities?: { entity: string; type: string }[];
  sentiment?: 'positive' | 'negative' | 'neutral';
  suggested_followups?: string[];
  internal_links?: { source_node_id: string; target_node_id: string; relationship: string }[];
  summary?: string;
}

const STRUCTURED_SYSTEM_PROMPT = `
Your task is to respond to the user's query based on the provided conversation history.
You MUST structure your entire response as a single JSON object conforming to the following TypeScript interface:

interface StructuredLLMOutput {
  main_response: string; // The primary natural language answer to the user's last message.
  identified_topics?: string[]; // List of key topics discussed in the main_response.
  key_entities?: { entity: string; type: string }[]; // List of named entities (PERSON, ORG, LOC, etc.) mentioned.
  sentiment?: 'positive' | 'negative' | 'neutral'; // Overall sentiment of the main_response.
  suggested_followups?: string[]; // 1-3 relevant follow-up questions the user might ask.
  internal_links?: { source_node_id: string; target_node_id: string; relationship: string }[]; // Placeholder for potential future linking logic within the context graph. Leave empty for now.
  summary?: string; // A very brief (1-2 sentence) summary of the main_response content.
}

Ensure your output is ONLY the valid JSON object, starting with { and ending with }. Do not include any text before or after the JSON object.
`;

export async function handleLLMRequest(req: Request, res: Response) {
  try {
    const { messages: originalMessages, model } = req.body as { messages: Message[], model: string };

    if (!model || typeof model !== 'string') {
      return res.status(400).json({ message: 'Model identifier is required' });
    }
    if (!originalMessages || !Array.isArray(originalMessages)) {
      return res.status(400).json({ message: 'Valid messages array is required' });
    }

    // Parse provider and modelName
    const parts = model.split(':');
    if (parts.length !== 2) {
        return res.status(400).json({ message: 'Invalid model format. Use "provider:model_name".' });
    }
    const [provider, modelName] = parts;

    // Prepend the structured system prompt
    const messagesWithSystemPrompt: Message[] = [
        { role: 'system', content: STRUCTURED_SYSTEM_PROMPT },
        ...originalMessages
    ];

    let responseData: any;
    let apiKey: string | undefined;

    switch (provider.toLowerCase()) {
      case 'openai':
      case 'o':
        apiKey = process.env.OPENAI_API_KEY;
        if (!apiKey) throw new Error('OpenAI API key not configured in Replit Secrets.');
        responseData = await callOpenAI(messagesWithSystemPrompt, apiKey, modelName);
        break;
      case 'anthropic':
      case 'a':
        apiKey = process.env.ANTHROPIC_API_KEY;
        if (!apiKey) throw new Error('Anthropic API key not configured in Replit Secrets.');
        // Anthropic might need message format adjustments (e.g., no system message first)
        // For simplicity now, we send it, but this might need refinement.
        // Check Anthropic docs for best way to handle system prompts.
        responseData = await callAnthropic(messagesWithSystemPrompt, apiKey, modelName);
        break;
      case 'gemini':
      case 'g':
        apiKey = process.env.GEMINI_API_KEY;
        if (!apiKey) throw new Error('Google Gemini API key not configured in Replit Secrets.');
         // Gemini often handles system prompts differently or not at all in older APIs.
         // Adjust message format if needed for Gemini. May need to integrate system prompt
         // into the first user message or use specific API features if available.
         // Let's try sending it as is for now.
        responseData = await callGemini(messagesWithSystemPrompt, apiKey, modelName);
        break;
      default:
        return res.status(400).json({ message: `Unsupported provider: ${provider}` });
    }

     // --- Attempt to Parse Structured Output ---
     let finalResponse: { message: StructuredLLMOutput | Message } = { message: responseData?.choices?.[0]?.message || responseData?.content?.[0] || { role: 'assistant', content: 'Error: Could not extract response.' } };
     let rawContent = finalResponse.message.content;

     if (typeof rawContent === 'string') {
        try {
            // Attempt to parse the string content as JSON
            const parsedJson: StructuredLLMOutput = JSON.parse(rawContent);
            // Basic validation: check for main_response
            if (parsedJson && typeof parsedJson.main_response === 'string') {
                 finalResponse = { message: parsedJson }; // Use the parsed JSON object
            } else {
                 console.warn("LLM response was not valid JSON or lacked main_response. Falling back to raw text.");
                 finalResponse = { message: { role: 'assistant', content: rawContent } }; // Fallback
            }
        } catch (parseError) {
            console.warn("Failed to parse LLM response as JSON. Falling back to raw text. Error:", parseError);
            finalResponse = { message: { role: 'assistant', content: rawContent } }; // Fallback
        }
     } else {
        // If content is not a string, wrap it in a standard message structure
        finalResponse = { message: { role: 'assistant', content: JSON.stringify(rawContent) || 'Received non-string content.' } };
     }


    res.json(finalResponse); // Send the potentially structured message back

  } catch (error) {
    console.error(`Error in handleLLMRequest for model ${req.body?.model}:`, error);
    res.status(500).json({
      message: 'Internal server error processing LLM request',
      error: error instanceof Error ? error.message : 'Unknown error'
    });
  }
}
