        /**
         * Basic Event Bus for cross-module communication.
         */
        class EventBus {
            constructor() {
                this.listeners = {};
            }

            // Add a listener for a specific event
            subscribe(event, callback) {
                if (!this.listeners[event]) {
                    this.listeners[event] = [];
                }
                this.listeners[event].push(callback);
            }

            // Remove a listener for a specific event
            unsubscribe(event, callback) {
                if (!this.listeners[event]) return;
                this.listeners[event] = this.listeners[event].filter(listener => listener !== callback);
            }

            // Trigger an event, calling all its listeners
            publish(event, data) {
                if (!this.listeners[event]) return;
                this.listeners[event].forEach(callback => {
                    try {
                        callback(data);
                    } catch (error) {
                        console.error(`Error in event listener for ${event}:`, error);
                    }
                });
            }
        }

        /**
         * Manages the Three.js scene, camera, renderer, and basic environment.
         */
        class SceneManager {
            constructor(containerId) {
                this.container = document.getElementById(containerId);
                if (!this.container) {
                    console.error(`Container element with ID "${containerId}" not found.`);
                    return;
                }
                this.width = this.container.clientWidth;
                this.height = this.container.clientHeight;

                this.setupScene();
                this.setupCamera();
                this.setupRenderer();
                this.setupLights();
                this.setupGrid();
                this.setupControls(); // Setup OrbitControls

                // Bind 'this' for the event listener
                this.onWindowResize = this.onWindowResize.bind(this);
                window.addEventListener('resize', this.onWindowResize);
            }

            // Initialize the Three.js scene
            setupScene() {
                this.scene = new THREE.Scene();
                this.scene.background = new THREE.Color(0x000811); // Dark blue background
                this.scene.fog = new THREE.FogExp2(0x000811, 0.0007); // Matching fog
            }

            // Initialize the perspective camera
            setupCamera() {
                this.camera = new THREE.PerspectiveCamera(
                    75, // Field of view
                    this.width / this.height, // Aspect ratio
                    0.1, // Near clipping plane
                    10000 // Far clipping plane
                );
                this.camera.position.set(0, 250, 1000); // Initial camera position
                this.camera.lookAt(0, 0, 0); // Look at the center of the scene
                this.initialCameraPosition = this.camera.position.clone(); // Store for reset
            }

            // Initialize the WebGL renderer
            setupRenderer() {
                this.renderer = new THREE.WebGLRenderer({ antialias: true }); // Enable anti-aliasing
                this.renderer.setSize(this.width, this.height);
                this.renderer.setPixelRatio(window.devicePixelRatio); // Adjust for high-DPI screens
                this.container.appendChild(this.renderer.domElement); // Add renderer to the DOM
            }

            // Add basic lighting to the scene
            setupLights() {
                const ambientLight = new THREE.AmbientLight(0x404040, 1); // Soft ambient light
                this.scene.add(ambientLight);

                const directionalLight = new THREE.DirectionalLight(0xffffff, 0.6); // White directional light
                directionalLight.position.set(500, 500, 500);
                this.scene.add(directionalLight);

                 const pointLight = new THREE.PointLight(0x00ffff, 0.8, 2000); // Cyan point light near center
                 pointLight.position.set(0, 100, 0);
                 this.scene.add(pointLight);
            }

            // Create the retro ground grid
            setupGrid() {
                const gridSize = 10000; // Larger grid
                const gridDivisions = 100; // More divisions
                const gridColorCenter = new THREE.Color(0x00ffff); // Cyan
                const gridColorGrid = new THREE.Color(0xff00ff); // Magenta

                const gridHelper = new THREE.GridHelper(gridSize, gridDivisions, gridColorCenter, gridColorGrid);
                gridHelper.position.y = -300; // Position grid lower
                this.scene.add(gridHelper);
            }

             // Setup camera orbit controls
            setupControls() {
                // Check if OrbitControls is loaded
                if (typeof THREE.OrbitControls === 'undefined') {
                    console.error("OrbitControls not loaded. Make sure the script is included.");
                    return;
                }
                this.controls = new THREE.OrbitControls(this.camera, this.renderer.domElement);
                this.controls.enableDamping = true; // Smooth camera movement
                this.controls.dampingFactor = 0.05;
                this.controls.screenSpacePanning = false; // Pan in plane parallel to camera
                this.controls.minDistance = 100; // Prevent zooming too close
                this.controls.maxDistance = 5000; // Prevent zooming too far
                this.controls.maxPolarAngle = Math.PI / 1.8; // Limit vertical rotation slightly
                this.controls.target.set(0, 0, 0); // Set initial target
                this.controls.update(); // Initial update
            }

            // Handle window resize events
            onWindowResize() {
                this.width = this.container.clientWidth;
                this.height = this.container.clientHeight;

                this.camera.aspect = this.width / this.height;
                this.camera.updateProjectionMatrix(); // Update camera projection

                this.renderer.setSize(this.width, this.height); // Update renderer size
            }

             // Reset camera to initial position and target
            resetView() {
                if (!this.controls) return;
                this.controls.reset(); // Resets target and position based on initial setup
                // Optionally force position if reset() doesn't work as expected
                // this.camera.position.copy(this.initialCameraPosition);
                // this.controls.target.set(0, 0, 0);
                this.controls.update();
            }


            // Render the scene
            render() {
                 if (this.controls) {
                    this.controls.update(); // Update controls in the render loop for damping
                 }
                this.renderer.render(this.scene, this.camera);
            }
        }

        /**
         * Manages the main context bubble sphere and its expansion.
         */
        class ContextBubble {
            constructor(scene, eventBus) {
                this.scene = scene;
                this.eventBus = eventBus;

                this.baseRadius = 500;
                this.currentRadius = this.baseRadius;
                this.maxRadiusBeforeExpand = this.baseRadius * 1.4; // When to trigger expansion visually
                this.expansionFactor = 1.5;
                this.bubbles = []; // Store current and ghost bubbles
                this.nodeCount = 0; // Track number of nodes

                this.createInitialBubble();

                // Listen for node additions to potentially trigger expansion
                this.eventBus.subscribe('node:created', () => {
                    this.nodeCount++;
                    this.checkExpansion();
                });
                 // Listen for context updates (placeholder for actual context size)
                 this.eventBus.subscribe('context:update', (contextSize) => {
                    // Example: Adjust visual radius based on simulated size
                    // This logic might need refinement based on how contextSize relates to desired radius
                    const targetRadius = this.baseRadius * Math.sqrt(Math.max(1, contextSize / 1000)); // Example scaling
                    this.targetVisualRadius = Math.min(targetRadius, this.maxRadiusBeforeExpand);
                 });

                 this.targetVisualRadius = this.baseRadius; // Target for smooth scaling
            }

            // Creates the material for the bubble mesh
            createBubbleMaterial(color, opacity, wireframe = false) {
                return new THREE.MeshBasicMaterial({
                    color: color,
                    transparent: true,
                    opacity: opacity,
                    wireframe: wireframe,
                    side: THREE.DoubleSide, // Render both sides
                    depthWrite: false // Prevent occluding objects inside
                });
            }

             // Creates the wireframe overlay for a bubble
            createWireframeOverlay(geometry, color, opacity) {
                const wireframeGeometry = new THREE.WireframeGeometry(geometry);
                const wireframeMaterial = new THREE.LineBasicMaterial({
                    color: color,
                    transparent: true,
                    opacity: opacity,
                    linewidth: 1 // Note: linewidth > 1 may not work on all platforms
                });
                return new THREE.LineSegments(wireframeGeometry, wireframeMaterial);
            }


            // Create the initial bubble
            createInitialBubble() {
                const geometry = new THREE.SphereGeometry(this.baseRadius, 32, 16); // Standard sphere
                // Main bubble mesh (slightly transparent fill)
                const bubbleMesh = new THREE.Mesh(geometry, this.createBubbleMaterial(0x003333, 0.05)); // Dark cyan fill

                // Wireframe overlay
                const wireframe = this.createWireframeOverlay(geometry, 0x00ffcc, 0.3); // Bright cyan wireframe

                bubbleMesh.add(wireframe); // Add wireframe as child
                this.scene.add(bubbleMesh);

                this.currentBubble = {
                    mesh: bubbleMesh,
                    wireframe: wireframe,
                    radius: this.baseRadius,
                    creation: Date.now()
                };
                this.bubbles.push(this.currentBubble);
            }

            // Check if the bubble needs to expand based on node count or other metrics
            checkExpansion() {
                 // Simple expansion trigger: expand every 10 nodes for demonstration
                const expansionThreshold = 10;
                if (this.nodeCount > 0 && this.nodeCount % expansionThreshold === 0) {
                     console.log(`Node count ${this.nodeCount} reached, triggering expansion.`);
                     this.expand();
                     this.eventBus.publish('bubble:expanded');
                 }

                 // Alternative: Check based on visual radius target exceeding threshold
                 // if (this.targetVisualRadius >= this.maxRadiusBeforeExpand && this.currentBubble.radius < this.maxRadiusBeforeExpand * 0.99) {
                 //     console.log("Visual radius target exceeds threshold, triggering expansion.");
                 //     this.expand();
                 //     this.eventBus.publish('bubble:expanded');
                 // }
            }


            // Expand the bubble, leaving a ghost behind
             expand() {
                if (!this.currentBubble) return;

                const oldRadius = this.currentBubble.radius;
                const oldMesh = this.currentBubble.mesh;
                const oldWireframe = this.currentBubble.wireframe;

                // Make current bubble a ghost
                oldMesh.material.color.setHex(0x227766); // Fade color
                oldMesh.material.opacity = 0.03; // Fade opacity
                oldWireframe.material.color.setHex(0x227766);
                oldWireframe.material.opacity = 0.1;

                // Create new, larger bubble
                const newRadius = oldRadius * this.expansionFactor;
                const newGeometry = new THREE.SphereGeometry(newRadius, 32, 16);

                // New main mesh
                const newBubbleMesh = new THREE.Mesh(newGeometry, this.createBubbleMaterial(0x003333, 0.05));

                // New wireframe
                const newWireframe = this.createWireframeOverlay(newGeometry, 0x00ffcc, 0.3);
                newBubbleMesh.add(newWireframe);
                this.scene.add(newBubbleMesh);

                // Update current bubble reference
                this.currentBubble = {
                    mesh: newBubbleMesh,
                    wireframe: newWireframe,
                    radius: newRadius,
                    creation: Date.now()
                };
                this.bubbles.push(this.currentBubble); // Add new bubble to the list

                 // Update thresholds and target radius
                 this.baseRadius = newRadius; // The new bubble becomes the base for future scaling
                 this.maxRadiusBeforeExpand = this.baseRadius * 1.4;
                 this.targetVisualRadius = this.baseRadius; // Reset visual target

                 console.log(`Bubble expanded to radius: ${newRadius.toFixed(0)}`);
            }


            // Update bubble appearance (fading, wobble)
            update() {
                const now = Date.now();
                const wobbleSpeed = 0.0005;
                const wobbleAmount = 0.01;

                // Smoothly scale the current bubble towards the target visual radius
                if (this.currentBubble) {
                    const currentScale = this.currentBubble.mesh.scale.x;
                    const targetScale = this.targetVisualRadius / this.currentBubble.radius;
                    const newScale = THREE.MathUtils.lerp(currentScale, targetScale, 0.05); // Smoothing factor
                    this.currentBubble.mesh.scale.set(newScale, newScale, newScale);
                }


                // Fade out and remove old bubbles
                this.bubbles = this.bubbles.filter(bubble => {
                    if (bubble === this.currentBubble) {
                         // Apply wobble to current bubble's wireframe for visual interest
                         if (bubble.wireframe) {
                             const wobble = Math.sin(now * wobbleSpeed) * wobbleAmount + 1;
                             // Apply wobble subtly, maybe just to wireframe scale?
                             // bubble.wireframe.scale.set(wobble, 1, wobble); // Can cause visual artifacts if mesh doesn't scale
                         }
                        return true; // Keep current bubble
                    }

                    const age = now - bubble.creation;
                    const maxAge = 20000; // 20 seconds lifespan for ghosts

                    if (age > maxAge) {
                        // Remove from scene and dispose resources
                        if (bubble.mesh) this.scene.remove(bubble.mesh);
                        if (bubble.wireframe) this.scene.remove(bubble.wireframe); // Ensure wireframe is also removed if separate
                        bubble.mesh?.geometry?.dispose();
                        bubble.mesh?.material?.dispose();
                        bubble.wireframe?.geometry?.dispose();
                        bubble.wireframe?.material?.dispose();
                        return false; // Remove from array
                    } else {
                        // Fade out ghost bubbles
                        const fadeFactor = 1.0 - (age / maxAge);
                         if (bubble.mesh?.material) {
                             bubble.mesh.material.opacity = 0.03 * fadeFactor;
                         }
                         if (bubble.wireframe?.material) {
                             bubble.wireframe.material.opacity = 0.1 * fadeFactor;
                         }
                        return true; // Keep fading bubble
                    }
                });
            }
        }

        /**
         * Manages the creation and layout of topic nodes.
         */
        class NodeSystem {
            constructor(scene, eventBus, contextBubble) {
                this.scene = scene;
                this.eventBus = eventBus;
                this.contextBubble = contextBubble; // Reference to the bubble system

                this.nodes = [];
                this.nodeIdCounter = 0;

                // Define colors for different topic categories
                this.topicColors = {
                    technology: 0x00ff88, // Mint green
                    science: 0x44aaff,    // Sky blue
                    history: 0xffaa22,    // Orange
                    art: 0xff55dd,        // Pink
                    philosophy: 0xeeeeee, // Light grey
                    gaming: 0xff4444,     // Red
                    music: 0xcc66ff,      // Purple
                    default: 0xaaaaaa     // Grey
                };

                // Force-directed layout parameters
                this.repulsionForce = 80; // Increased repulsion
                this.attractionForce = 0.005; // Lower attraction
                this.centerAttraction = 0.001; // Pull towards center slightly
                this.bubbleBoundaryForce = 0.1; // Force to stay inside bubble
                this.maxSpeed = 5; // Max movement speed per frame
                this.damping = 0.95; // Slow down movement over time

                this.setupGeometries();

                 // Listen for request to get a random node ID (for simulation)
                 this.eventBus.subscribe('request:randomNodeId', () => {
                    if (this.nodes.length > 0) {
                        const randomIndex = Math.floor(Math.random() * this.nodes.length);
                        return this.nodes[randomIndex].id;
                    }
                    return null; // No nodes exist yet
                 });
            }

            // Pre-create geometries for efficiency
            setupGeometries() {
                this.nodeGeometry = new THREE.IcosahedronGeometry(15, 0); // Low-poly node shape
                this.glowGeometry = new THREE.IcosahedronGeometry(15 * 1.3, 0); // Slightly larger for glow
            }

            // Create a new node based on topic data
            createNode(topicData) {
                const id = `node_${this.nodeIdCounter++}`;
                const category = topicData.category || 'default';
                const color = this.topicColors[category] || this.topicColors.default;
                const emissiveColor = new THREE.Color(color).multiplyScalar(0.3); // Dim glow

                // --- Calculate Position ---
                // Place node initially on the *current* bubble's surface
                const bubbleRadius = this.contextBubble.currentBubble.radius * 0.95; // Slightly inside current bubble scale

                // Random spherical coordinates
                const phi = Math.acos(-1 + (2 * Math.random())); // inclination (0 to PI)
                const theta = Math.random() * Math.PI * 2; // azimuth (0 to 2PI)

                const x = bubbleRadius * Math.sin(phi) * Math.cos(theta);
                const y = bubbleRadius * Math.sin(phi) * Math.sin(theta);
                const z = bubbleRadius * Math.cos(phi);

                // --- Create Node Mesh ---
                const material = new THREE.MeshPhongMaterial({
                    color: color,
                    emissive: emissiveColor, // Make it glow slightly
                    shininess: 50, // Moderate shininess
                    specular: 0x555555 // Grey specular highlight
                });
                const nodeMesh = new THREE.Mesh(this.nodeGeometry, material);
                nodeMesh.position.set(x, y, z);
                nodeMesh.userData = { id: id, topic: topicData.topic }; // Store data

                // --- Create Glow Effect (Optional but nice) ---
                // Using a slightly larger, transparent mesh
                const glowMaterial = new THREE.MeshBasicMaterial({
                    color: color,
                    transparent: true,
                    opacity: 0.2,
                    side: THREE.BackSide // Render back face for halo effect
                });
                const glowMesh = new THREE.Mesh(this.glowGeometry, glowMaterial);
                // glowMesh.scale.multiplyScalar(1.3); // Already using larger geometry
                nodeMesh.add(glowMesh); // Add glow as a child

                this.scene.add(nodeMesh);

                // --- Track Node Data ---
                const newNode = {
                    id: id,
                    mesh: nodeMesh,
                    glow: glowMesh,
                    topic: topicData.topic,
                    category: category,
                    connectedTo: [], // IDs of connected nodes
                    velocity: new THREE.Vector3(0, 0, 0),
                    force: new THREE.Vector3(0, 0, 0),
                    targetPosition: nodeMesh.position.clone() // For smoother movement later if needed
                };
                this.nodes.push(newNode);

                console.log(`Node created: ${id} (${topicData.topic}) at radius ${bubbleRadius.toFixed(0)}`);
                this.eventBus.publish('node:created', newNode); // Notify other systems
                return id;
            }

            // Get a node object by its ID
            getNodeById(id) {
                return this.nodes.find(node => node.id === id);
            }

            // Register a connection between two nodes (called by EdgeSystem)
            connectNodes(fromId, toId) {
                const fromNode = this.getNodeById(fromId);
                const toNode = this.getNodeById(toId);

                if (fromNode && toNode) {
                    // Avoid duplicate connections
                    if (!fromNode.connectedTo.includes(toId)) {
                        fromNode.connectedTo.push(toId);
                    }
                    if (!toNode.connectedTo.includes(fromId)) {
                        toNode.connectedTo.push(fromId);
                    }
                } else {
                     console.warn(`Attempted to connect non-existent nodes: ${fromId} to ${toId}`);
                }
            }

            // Update node positions based on force simulation
            update() {
                 if (this.nodes.length < 2) return; // No forces needed for 0 or 1 node

                 const currentBubbleRadius = this.contextBubble.currentBubble.radius * 0.95; // Target radius

                // Calculate forces for each node
                for (let i = 0; i < this.nodes.length; i++) {
                    const nodeA = this.nodes[i];
                    nodeA.force.set(0, 0, 0); // Reset force for this frame

                    // 1. Repulsion from other nodes
                    for (let j = i + 1; j < this.nodes.length; j++) {
                        const nodeB = this.nodes[j];
                        const diff = new THREE.Vector3().subVectors(nodeA.mesh.position, nodeB.mesh.position);
                        const distanceSq = diff.lengthSq(); // Use squared distance for efficiency

                        if (distanceSq > 1e-6 && distanceSq < (this.repulsionForce * 10)**2) { // Only calculate within a range
                             const distance = Math.sqrt(distanceSq);
                             // Force = direction * magnitude (stronger when closer)
                             const magnitude = this.repulsionForce / distanceSq; // Inverse square law
                             const repulsion = diff.normalize().multiplyScalar(magnitude);
                             nodeA.force.add(repulsion);
                             nodeB.force.sub(repulsion); // Apply equal and opposite force to B
                        }
                    }

                    // 2. Attraction to connected nodes (Spring force)
                    for (const connectedId of nodeA.connectedTo) {
                        const connectedNode = this.getNodeById(connectedId);
                        if (connectedNode) {
                             // Only calculate attraction once per pair (e.g., when i < connectedNode's index)
                             // This requires finding the index, might be simpler to calculate both ways and halve the force
                            const diff = new THREE.Vector3().subVectors(connectedNode.mesh.position, nodeA.mesh.position);
                            const distance = diff.length();
                            const idealDistance = 150; // Desired distance between connected nodes
                            // Force = direction * magnitude (pulls together/pushes apart to ideal distance)
                            const displacement = distance - idealDistance;
                            const magnitude = this.attractionForce * displacement; // Linear spring force
                            const attraction = diff.normalize().multiplyScalar(magnitude);
                            nodeA.force.add(attraction);
                        }
                    }

                    // 3. Attraction towards the center (keeps the graph centered)
                    const centerForce = nodeA.mesh.position.clone().multiplyScalar(-this.centerAttraction);
                    nodeA.force.add(centerForce);


                    // 4. Boundary force (keep nodes inside the current bubble)
                     const distanceFromCenter = nodeA.mesh.position.length();
                     if (distanceFromCenter > currentBubbleRadius) {
                         const directionToCenter = nodeA.mesh.position.clone().normalize().multiplyScalar(-1);
                         // Force increases the further outside the node is
                         const boundaryMagnitude = this.bubbleBoundaryForce * (distanceFromCenter - currentBubbleRadius);
                         const boundaryPush = directionToCenter.multiplyScalar(boundaryMagnitude);
                         nodeA.force.add(boundaryPush);
                     }
                }

                // Update positions based on forces
                for (const node of this.nodes) {
                    // Apply force to velocity
                    node.velocity.add(node.force);

                    // Apply damping (friction)
                    node.velocity.multiplyScalar(this.damping);

                    // Limit speed
                    if (node.velocity.lengthSq() > this.maxSpeed * this.maxSpeed) {
                        node.velocity.normalize().multiplyScalar(this.maxSpeed);
                    }

                    // Update position
                    node.mesh.position.add(node.velocity);

                    // Simple rotation for visual effect
                    node.mesh.rotation.x += 0.005 + node.velocity.y * 0.001;
                    node.mesh.rotation.y += 0.005 + node.velocity.x * 0.001;
                }
            }
        }

        /**
         * Manages the visual edges (lines) connecting nodes.
         */
        class EdgeSystem {
            constructor(scene, nodeSystem, eventBus) {
                this.scene = scene;
                this.nodeSystem = nodeSystem;
                this.eventBus = eventBus;

                this.edges = []; // Stores { fromId, toId, line }
                this.lineMaterial = new THREE.LineBasicMaterial({
                    color: 0xff00ff, // Magenta color for edges
                    transparent: true,
                    opacity: 0.4, // More subtle edges
                    linewidth: 1 // Note: May not render thicker on all systems
                });
            }

            // Create a visual edge between two nodes
            createEdge(fromNodeId, toNodeId) {
                // Prevent creating duplicate edges
                const existingEdge = this.edges.find(edge =>
                    (edge.fromId === fromNodeId && edge.toId === toNodeId) ||
                    (edge.fromId === toNodeId && edge.toId === fromNodeId)
                );
                if (existingEdge) {
                    console.warn(`Edge between ${fromNodeId} and ${toNodeId} already exists.`);
                    return null;
                }


                const fromNode = this.nodeSystem.getNodeById(fromNodeId);
                const toNode = this.nodeSystem.getNodeById(toNodeId);

                if (!fromNode || !toNode) {
                    console.error(`Cannot create edge: Node not found (from: ${fromNodeId}, to: ${toNodeId})`);
                    return null;
                }

                // Create line geometry with initial points
                const points = [
                    fromNode.mesh.position.clone(),
                    toNode.mesh.position.clone()
                ];
                const geometry = new THREE.BufferGeometry().setFromPoints(points);

                // Create the line mesh
                const line = new THREE.Line(geometry, this.lineMaterial.clone()); // Clone material if properties might change per line

                this.scene.add(line);

                // Register the connection in the NodeSystem
                this.nodeSystem.connectNodes(fromNodeId, toNodeId);

                // Track the edge
                const newEdge = {
                    fromId: fromNodeId,
                    toId: toNodeId,
                    line: line
                };
                this.edges.push(newEdge);

                console.log(`Edge created between ${fromNodeId} and ${toNodeId}`);
                return newEdge;
            }

            // Update edge positions to follow connected nodes
            update() {
                for (const edge of this.edges) {
                    const fromNode = this.nodeSystem.getNodeById(edge.fromId);
                    const toNode = this.nodeSystem.getNodeById(edge.toId);

                    // Check if nodes still exist (might be removed in future implementation)
                    if (fromNode && toNode) {
                        // Get the current positions array from the geometry
                        const positions = edge.line.geometry.attributes.position;

                        // Update the first point (fromNode)
                        positions.setXYZ(0, fromNode.mesh.position.x, fromNode.mesh.position.y, fromNode.mesh.position.z);
                        // Update the second point (toNode)
                        positions.setXYZ(1, toNode.mesh.position.x, toNode.mesh.position.y, toNode.mesh.position.z);

                        // Mark the position attribute as needing an update
                        positions.needsUpdate = true;

                        // Optional: Update bounding sphere for frustum culling
                        edge.line.geometry.computeBoundingSphere();
                    } else {
                        // If a node is missing, remove the edge (or handle appropriately)
                        console.warn(`Node missing for edge ${edge.fromId}-${edge.toId}. Removing edge.`);
                        this.scene.remove(edge.line);
                        edge.line.geometry.dispose();
                        // Remove edge from the array (needs careful handling while iterating)
                        // Consider marking for removal and cleaning up after the loop
                    }
                }
                 // Cleanup edges marked for removal if implemented
            }
        }


        /**
         * Manages the AI "drones" that traverse the knowledge graph.
         */
        class AgentSystem {
            constructor(scene, nodeSystem, edgeSystem, eventBus) {
                this.scene = scene;
                this.nodeSystem = nodeSystem;
                this.edgeSystem = edgeSystem; // Needed to find paths? Maybe just nodeSystem connections are enough
                this.eventBus = eventBus;

                this.agents = []; // Stores agent objects
                this.agentSpeed = 2.5; // Units per second (adjust based on scene scale)
                this.trailLength = 30; // Number of points in the trail
                this.isVisible = true; // Controlled by UI toggle

                this.setupGeometries();

                 // Listen for bubble expansion to spawn agents
                 this.eventBus.subscribe('bubble:expanded', () => {
                     if (this.isVisible) {
                         this.spawnAgents(2); // Spawn 2 agents on expansion
                     }
                 });
                 // Listen for UI toggle
                 this.eventBus.subscribe('drones:toggle', (visible) => {
                     this.isVisible = visible;
                     this.agents.forEach(agent => {
                         agent.mesh.visible = visible;
                         agent.trail.visible = visible;
                     });
                     if (visible && this.agents.length === 0 && this.nodeSystem.nodes.length >= 2) {
                         // Spawn initial agents if toggled on and none exist
                         this.spawnAgents(3);
                     }
                 });
            }

            // Pre-create agent geometry
            setupGeometries() {
                // Simple pyramid shape for the drone
                this.agentGeometry = new THREE.ConeGeometry(6, 18, 4); // Base radius, height, segments (makes pyramid)
                this.agentGeometry.rotateX(Math.PI / 2); // Point forward
            }

            // Spawn a specified number of agents
            spawnAgents(count = 1) {
                 if (!this.isVisible) return; // Don't spawn if hidden

                for (let i = 0; i < count; i++) {
                    if (this.nodeSystem.nodes.length < 1) continue; // Need at least one node

                    // Pick a random starting node
                    const startNodeIndex = Math.floor(Math.random() * this.nodeSystem.nodes.length);
                    const startNode = this.nodeSystem.nodes[startNodeIndex];

                    // --- Create Agent Mesh ---
                    const material = new THREE.MeshBasicMaterial({
                    color: 0x55ffff, // Light cyan/aqua color
                    wireframe: true // Wireframe look
                });
                const agentMesh = new THREE.Mesh(this.agentGeometry, material);
                agentMesh.position.copy(startNode.mesh.position); // Start at the node
                this.scene.add(agentMesh);

                // --- Create Trail System ---
                const trailMaterial = new THREE.LineBasicMaterial({
                    color: 0x55ffff,
                    transparent: true,
                    opacity: 0.4,
                    linewidth: 1,
                    blending: THREE.AdditiveBlending // Brighter where lines overlap
                });
                const trailGeometry = new THREE.BufferGeometry();
                // Initialize positions with the agent's start position
                const trailPositions = new Float32Array(this.trailLength * 3);
                for(let j=0; j < this.trailLength; ++j) {
                    trailPositions[j*3 + 0] = agentMesh.position.x;
                    trailPositions[j*3 + 1] = agentMesh.position.y;
                    trailPositions[j*3 + 2] = agentMesh.position.z;
                }
                trailGeometry.setAttribute('position', new THREE.BufferAttribute(trailPositions, 3));
                const trail = new THREE.Line(trailGeometry, trailMaterial);
                this.scene.add(trail);

                // --- Add Agent Data ---
                this.agents.push({
                    mesh: agentMesh,
                    trail: trail,
                    trailPositions: trailPositions, // Reference to the Float32Array
                    currentNodeId: startNode.id,
                    targetNodeId: null, // ID of the target node
                    progress: 0, // 0 to 1 progress along the current edge
                    state: 'orbiting', // 'orbiting', 'traveling'
                    orbitParams: { // Parameters for orbiting behavior
                        radius: 25 + Math.random() * 15,
                        speed: (Math.random() * 0.02 + 0.01) * (Math.random() < 0.5 ? 1 : -1), // Random speed and direction
                        angle: Math.random() * Math.PI * 2
                    }
                });
                 console.log(`Agent spawned at node ${startNode.id}`);
            }
        }

            // Find a suitable next node for an agent to travel to
            findNextNode(agent) {
                const currentNode = this.nodeSystem.getNodeById(agent.currentNodeId);
                if (!currentNode || currentNode.connectedTo.length === 0) {
                    // No connections or node missing, pick any other random node
                    if (this.nodeSystem.nodes.length < 2) return null; // Cannot find another node
                    let randomIndex;
                    do {
                        randomIndex = Math.floor(Math.random() * this.nodeSystem.nodes.length);
                    } while (this.nodeSystem.nodes[randomIndex].id === agent.currentNodeId); // Ensure it's not the same node
                    return this.nodeSystem.nodes[randomIndex].id;
                }

                // Pick a random connected node, preferably one not just visited (if we tracked history)
                const connectedIndex = Math.floor(Math.random() * currentNode.connectedTo.length);
                return currentNode.connectedTo[connectedIndex];
            }

            // Update agent positions, states, and trails
            update(deltaTime) { // deltaTime in seconds
                 if (!this.isVisible) return; // Skip updates if hidden

                const now = Date.now(); // For orbiting calculation

                this.agents.forEach(agent => {
                    const currentNode = this.nodeSystem.getNodeById(agent.currentNodeId);
                    if (!currentNode) {
                        // Current node doesn't exist? Maybe respawn agent?
                        // For now, just skip update for this agent
                        console.warn(`Agent's current node ${agent.currentNodeId} not found.`);
                        // TODO: Handle agent removal or respawn
                        return;
                    }

                    // --- State Machine ---
                    if (agent.state === 'orbiting') {
                        // Orbit around the current node
                        agent.orbitParams.angle += agent.orbitParams.speed; // Update orbit angle

                        const orbitPos = new THREE.Vector3(
                            Math.cos(agent.orbitParams.angle) * agent.orbitParams.radius,
                            Math.sin(agent.orbitParams.angle * 0.7) * agent.orbitParams.radius * 0.5, // Add some vertical variation
                            Math.sin(agent.orbitParams.angle) * agent.orbitParams.radius
                        );

                        // Add orbit offset to the node's current position
                        agent.mesh.position.copy(currentNode.mesh.position).add(orbitPos);
                        agent.mesh.lookAt(currentNode.mesh.position); // Look towards the node

                        // Randomly decide to travel to a new node
                        if (Math.random() < 0.005) { // Low probability each frame
                            const nextNodeId = this.findNextNode(agent);
                            if (nextNodeId) {
                                agent.targetNodeId = nextNodeId;
                                agent.state = 'traveling';
                                agent.progress = 0;
                                console.log(`Agent ${agent.mesh.uuid} starting travel from ${agent.currentNodeId} to ${agent.targetNodeId}`);
                            }
                        }

                    } else if (agent.state === 'traveling') {
                        const targetNode = this.nodeSystem.getNodeById(agent.targetNodeId);
                        if (!targetNode) {
                            // Target node disappeared? Revert to orbiting current node
                            console.warn(`Agent's target node ${agent.targetNodeId} not found. Reverting to orbit.`);
                            agent.state = 'orbiting';
                            agent.targetNodeId = null;
                            return; // Skip rest of travel logic for this frame
                        }

                        const startPos = currentNode.mesh.position;
                        const endPos = targetNode.mesh.position;
                        const totalDistance = startPos.distanceTo(endPos);

                        if (totalDistance < 1) { // Prevent division by zero or tiny distances
                             agent.progress = 1; // Consider arrived
                        } else {
                            // Calculate distance to move this frame
                            const distanceToMove = this.agentSpeed * (deltaTime || (1/60)); // Use deltaTime if available, else assume 60fps
                            agent.progress += distanceToMove / totalDistance;
                            agent.progress = Math.min(agent.progress, 1.0); // Clamp progress to 1
                        }


                        // Interpolate position using progress
                        agent.mesh.position.lerpVectors(startPos, endPos, agent.progress);

                        // Look towards the target node
                        if (totalDistance > 1) {
                            agent.mesh.lookAt(endPos);
                        }

                        // Check if arrived
                        if (agent.progress >= 1.0) {
                            console.log(`Agent ${agent.mesh.uuid} arrived at ${agent.targetNodeId}`);
                            agent.currentNodeId = agent.targetNodeId;
                            agent.targetNodeId = null;
                            agent.state = 'orbiting'; // Start orbiting the new node
                            // Reset orbit parameters for variety
                            agent.orbitParams.radius = 25 + Math.random() * 15;
                            agent.orbitParams.speed = (Math.random() * 0.02 + 0.01) * (Math.random() < 0.5 ? 1 : -1);
                            agent.orbitParams.angle = Math.random() * Math.PI * 2;
                        }
                    }

                    // --- Update Trail ---
                    this.updateTrail(agent);
                });

                 // TODO: Add logic to remove agents if they become orphaned or based on other criteria
            }

            // Update the points in an agent's trail line
            updateTrail(agent) {
                const positions = agent.trailPositions;

                // Shift all existing points back one position
                for (let i = this.trailLength - 1; i > 0; i--) {
                    positions[i * 3 + 0] = positions[(i - 1) * 3 + 0];
                    positions[i * 3 + 1] = positions[(i - 1) * 3 + 1];
                    positions[i * 3 + 2] = positions[(i - 1) * 3 + 2];
                }

                // Add the agent's current position to the front of the array
                positions[0] = agent.mesh.position.x;
                positions[1] = agent.mesh.position.y;
                positions[2] = agent.mesh.position.z;

                // Mark the geometry attribute as needing an update
                agent.trail.geometry.attributes.position.needsUpdate = true;
                // Optional: recompute bounding sphere for culling
                // agent.trail.geometry.computeBoundingSphere();
            }
        }

                try {
                    const apiKey = this.uiController.getApiKey();
                    if (!apiKey) {
                        throw new Error("API key not set");
                    }

                    // Prepare messages including the system prompt
                    const messages = [systemPrompt, ...this.conversationHistory];

                    const response = await fetch('https://api.openai.com/v1/chat/completions', {
                        method: 'POST',
                        headers: {
                            'Content-Type': 'application/json',
                            'Authorization': `Bearer ${apiKey}`
                        },
                        body: JSON.stringify({
                            model: this.currentModel,
                            messages: messages,
                            temperature: 0.7,
                            response_format: { type: "json_object" }, // Request JSON response
                            max_tokens: 1000
                        })
                    });

                    if (!response.ok) {
                        const errorData = await response.json();
                        throw new Error(errorData.error?.message || 'Failed to get response from OpenAI');
                    }

                    const data = await response.json();
                    const aiResponse = JSON.parse(data.choices[0].message.content);

                    // Add AI response to conversation history
                    this.conversationHistory.push({
                        role: "assistant",
                        content: aiResponse.response
                    });

                    // Publish the response
                    this.eventBus.publish('ai:response', aiResponse.response);
                    
                    // Publish structured data for visualization
                    this.eventBus.publish('response:structured', aiResponse);

                } catch (error) {
                    console.error("Error calling OpenAI API:", error);
                    this.eventBus.publish('ai:response', `Error: ${error.message}`);
                } finally {
                    this.isProcessing = false;
                    this.eventBus.publish('ai:processing', false);
                }
            }
        }

        /**
         * Enhanced NodeSystem with question nodes and text snippets
         */
        class NodeSystem {
            constructor(scene, eventBus, contextBubble) {
                this.scene = scene;
                this.eventBus = eventBus;
                this.contextBubble = contextBubble;
                this.fontLoader = new THREE.FontLoader();
                this.font = null;
                this.loadingFont = true;

                // Load font for text snippets
                this.fontLoader.load('https://cdn.jsdelivr.net/npm/three@0.128.0/examples/fonts/helvetiker_regular.typeface.json', 
                    (font) => {
                        this.font = font;
                        this.loadingFont = false;
                        console.log("Font loaded successfully");
                    },
                    undefined,
                    (error) => {
                        console.error("Error loading font:", error);
                    }
                );

                // Define colors for different types
                this.nodeColors = {
                    topic: {
                        technology: 0x00ff88,
                        science: 0x44aaff,
                        history: 0xffaa22,
                        art: 0xff55dd,
                        philosophy: 0xeeeeee,
                        gaming: 0xff4444,
                        music: 0xcc66ff,
                        default: 0xaaaaaa
                    },
                    question: 0xff3333, // Red for questions
                    snippet: 0x33ff33  // Green for text snippets
                };

                // ...rest of constructor
            }

            createTextSnippet(text, position) {
                if (this.loadingFont) {
                    console.warn("Font not loaded yet, cannot create text snippet");
                    return null;
                }

                const textGeometry = new THREE.TextGeometry(text, {
                    font: this.font,
                    size: 8,
                    height: 1,
                    curveSegments: 4,
                    bevelEnabled: false
                });

                const textMaterial = new THREE.MeshBasicMaterial({ 
                    color: this.nodeColors.snippet,
                    transparent: true,
                    opacity: 0.9
                });

                const textMesh = new THREE.Mesh(textGeometry, textMaterial);
                
                // Center the text
                textGeometry.computeBoundingBox();
                const textWidth = textGeometry.boundingBox.max.x - textGeometry.boundingBox.min.x;
                textMesh.position.set(position.x - textWidth/2, position.y, position.z);
                
                // Add a background plane
                const backgroundGeometry = new THREE.PlaneGeometry(textWidth + 20, 30);
                const backgroundMaterial = new THREE.MeshBasicMaterial({
                    color: 0x000000,
                    side: THREE.DoubleSide,
                    transparent: true,
                    opacity: 0.7
                });
                const background = new THREE.Mesh(backgroundGeometry, backgroundMaterial);
                background.position.set(position.x, position.y - 5, position.z - 1); // Behind the text
                textMesh.add(background);

                this.scene.add(textMesh);
                return textMesh;
            }

            createNode(nodeData) {
                const id = `node_${this.nodeIdCounter++}`;
                let color, emissiveColor;

                if (nodeData.type === 'question') {
                    color = this.nodeColors.question;
                } else if (nodeData.type === 'snippet') {
                    // For snippets, we'll use the text mesh instead of a sphere
                    const textMesh = this.createTextSnippet(nodeData.text, nodeData.position);
                    if (!textMesh) return null;
                    
                    return {
                        id: id,
                        mesh: textMesh,
                        type: 'snippet',
                        text: nodeData.text,
                        connectedTo: []
                    };
                } else {
                    // Topic node
                    const category = nodeData.category || 'default';
                    color = this.nodeColors.topic[category] || this.nodeColors.topic.default;
                }

                emissiveColor = new THREE.Color(color).multiplyScalar(0.3);

                // ...rest of node creation logic
            }
        }

        class ContextTracker {
            constructor(eventBus) {
                this.eventBus = eventBus;
                this.lastNodeId = null;
                this.lastNodeId = null; // Track the last added node for linking
                this.topicCategories = [
                    'technology', 'science', 'history', 'art',

                ];

                // Listen for processed AI responses
                this.eventBus.subscribe('response:processed', (data) => {
                    this.processResponse(data.text, data.userMessage);
                });
            }

            async processResponse(response, userMessage) {
                // In a real implementation, you might:
                // 1. Use NLP to extract entities/topics from the response
                // 2. Determine relationships between concepts
                // 3. Create nodes and edges based on the analysis

                // For this example, we'll simulate topic extraction with some simple rules
                this.contextSize += response.length; // Simple size tracking
                this.eventBus.publish('context:update', this.contextSize);

                // Randomly decide to add a new topic node based on the response
                if (Math.random() < 0.4) { // 40% chance to add a node per AI response
                    const category = this.topicCategories[Math.floor(Math.random() * this.topicCategories.length)];
                    const topicData = {
                        topic: this.extractMainTopic(response) || `Topic from conversation (${category})`,
                        category: category,
                        relatedTo: this.lastNodeId // Link to the previously added node
                    };
                    
                    console.log("Extracted new topic:", topicData);
                    this.eventBus.publish('topic:new', topicData);
                }
            }

            // Simple heuristic to extract a potential topic from text
            extractMainTopic(text) {
                // This is a very naive implementation - in a real app you'd use proper NLP
                const sentences = text.split(/[.!?]\s+/);
                if (sentences.length > 0) {
                    const firstSentence = sentences[0];
                    const words = firstSentence.split(/\s+/);
                    if (words.length > 3) {
                        // Try to find a noun phrase (very simplified)
                        return words.slice(0, 4).join(' ');
                    }
                    return firstSentence;
                }
                return null;
            }
        }

        /**
         * Orchestrates the entire visualization application.
         */
        class ContextBubbleApp {
            constructor() {
                console.log("Initializing ContextBubbleApp...");
                this.eventBus = new EventBus();

                // Core systems
                this.sceneManager = new SceneManager('visualization-container');
                if (!this.sceneManager.scene) return; // Stop if SceneManager failed

                // Visualization systems (pass dependencies)
                this.contextBubble = new ContextBubble(this.sceneManager.scene, this.eventBus);
                this.nodeSystem = new NodeSystem(this.sceneManager.scene, this.eventBus, this.contextBubble);
                this.edgeSystem = new EdgeSystem(this.sceneManager.scene, this.nodeSystem, this.eventBus);
                this.agentSystem = new AgentSystem(this.sceneManager.scene, this.nodeSystem, this.edgeSystem, this.eventBus);

                // Conversation systems with actual LLM integration
                this.uiController = new UIController(this.sceneManager, this.eventBus);
                this.chatInterface = new ChatInterface('chat-interface', 'user-input', 'send-button', 'chat-messages', this.eventBus);
                this.aiConnector = new AIConnector(this.eventBus, this.uiController);
                this.contextTracker = new ContextTracker(this.eventBus);

                // Clock for delta time calculation
                this.clock = new THREE.Clock();

                this.setupEventListeners();
                this.animate(); // Start the animation loop
                console.log("ContextBubbleApp Initialized Successfully.");

                 // Spawn initial agents if nodes exist (or wait for expansion)
                 setTimeout(() => { // Delay slightly to ensure nodes might be added
                     if (this.nodeSystem.nodes.length >= 2 && this.agentSystem.agents.length === 0) {
                         this.agentSystem.spawnAgents(3);
                     }
                 }, 2000);
            }

            // Centralized event listener setup
            setupEventListeners() {
                 // Handle new topic requests (from ContextTracker or UI simulation)
                 this.eventBus.subscribe('topic:new', (topicData) => {
                     const nodeId = this.nodeSystem.createNode(topicData);
                     if (nodeId && topicData.relatedTo) {
                         // Attempt to create an edge after a short delay to allow node positioning
                         setTimeout(() => {
                             this.edgeSystem.createEdge(topicData.relatedTo, nodeId);
                         }, 100); // Small delay
                     }
                 });

                 // Handle AI processing state to show/hide loading indicator
                 this.eventBus.subscribe('ai:processing', (isProcessing) => {
                     if (isProcessing) {
                         this.chatInterface.showLoadingIndicator();
                     } else {
                         this.chatInterface.hideLoadingIndicator();
                     }
                 });

                 // Handle AI responses to display in chat
                 this.eventBus.subscribe('ai:response', (response) => {
                     this.chatInterface.displayAIResponse(response);
                 });
            }

            // Main animation loop
            animate() {
                // Request the next frame
                requestAnimationFrame(this.animate.bind(this));

                // Calculate time delta for physics/animation updates
                const deltaTime = this.clock.getDelta();

                // Update systems that depend on time or physics
                this.nodeSystem.update(); // Update node physics/layout
                this.edgeSystem.update(); // Update edge positions
                this.agentSystem.update(deltaTime); // Update agent movement and trails
                this.contextBubble.update(); // Update bubble visuals (fading, wobble)

                // Render the scene
                this.sceneManager.render();
            }
        }
